package com.quantumparkour.todo;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;
import java.sql.Statement;

public class QuantumDatabase {

    private static final String DATABASE_URL = "jdbc:sqlite:quantumparkour.db";

    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(DATABASE_URL);
    }

    //TODO: REPLACE COINS WITH QWOBITS WHEN QWOBITS IS IMPLEMENTED
    public static void initialize() {
        try (Connection connection = getConnection(); Statement statement = connection.createStatement()) {
            String createPlayersTable = "CREATE TABLE IF NOT EXISTS Players (" +
                    "UUID TEXT UNIQUE NOT NULL, " +
                    "Username VARCHAR(16) NOT NULL, " +
                    "Coins INT NOT NULL, " +
                    "Rank VARCHAR(16) NOT NULL" +
                    ");";
            statement.execute(createPlayersTable);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}
